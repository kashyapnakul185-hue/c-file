Q2 - Employee ID & Name

#include <stdio.h>

struct employee {
    int id;
    char name[20];
    char design[25];
};

int main() {
    struct employee emp[50];
    printf("Enter details of employees:\\n");
    for(int i=0; i<1; i++){
        printf("\\nEmployee %d:\\n", i+1);
        printf("Enter ID: "); scanf("%d", &emp[i].id);
        printf("Enter Name: "); scanf("%s", emp[i].name);
        printf("Enter Designation: "); scanf("%s", emp[i].design);
    }
    printf("\\n| %-5s | %-25s |\\n", "ID", "Name");
    for(int i=0; i<1; i++){
        printf("| %-5d | %-25s |\\n", emp[i].id, emp[i].name);
    }
    return 0;
}
