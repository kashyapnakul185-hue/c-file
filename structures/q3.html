Q3 - Students Roll Sort

#include <stdio.h>

struct student {
    int roll;
    char name[20];
    int fee;
    char dob[20];
};

int main() {
    struct student stu[5], temp;
    printf("Enter details of 5 students:\\n");
    for(int i=0;i<5;i++){
        printf("\\nStudent %d:\\n", i+1);
        printf("Enter Roll Number: "); scanf("%d", &stu[i].roll);
        printf("Enter Name: "); scanf("%s", stu[i].name);
        printf("Enter Fees: "); scanf("%d", &stu[i].fee);
        printf("Enter DOB: "); scanf("%s", stu[i].dob);
    }
    for(int i=0;i<4;i++){
        for(int j=i+1;j<5;j++){
            if(stu[i].roll>stu[j].roll){
                temp=stu[i]; stu[i]=stu[j]; stu[j]=temp;
            }
        }
    }
    printf("\\n| %-10s | %-15s | %-10s | %-12s |\\n", "Roll No", "Name", "Fee", "DOB");
    for(int i=0;i<5;i++){
        printf("| %-10d | %-15s | %-10d | %-12s |\\n", stu[i].roll, stu[i].name, stu[i].fee, stu[i].dob);
    }
    return 0;
}
